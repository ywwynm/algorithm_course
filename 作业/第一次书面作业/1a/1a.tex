      \State $S \gets \text{空匹配集合}$
      \While{$\text{某一个男人 } m \text{ 还没有匹配} \textbf{ and } m \text{ 没有向所有女人都求过婚}$}

        \ElsIf{$w \text{ 认为 } m \text{ 比现在的伴侣 } m' \text{ 更好}$}
          \State $\text{从 } S \text{ 中移除配对 } w-m'$
          \State $\text{将配对 } w-m \text{ 添加到 } S$
        \EndIf
      \EndWhile
      \State
